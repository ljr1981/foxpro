	dbf_printer (a_file_name: STRING_8)
			-- A test-support only feature
		local
			l_file: RAW_FILE
			l_hex: STRING_32
			l_text: STRING_32
			i: INTEGER_32
		do
			create out_file.make_create_read_write (Out_file_name)
			create l_text.make_empty
			create l_file.make_open_read (a_file_name)
			line_out ("File-name: " + a_file_name + "%N")
			line_out ("======================================%N")
			line_out ("Legend:%N")
			line_out ("--------------------------------------%N")
			line_out (": Space-char%N")
			line_out ("======================================%N")
			line_out ("Hex-Addr=00-01-02-03-04-05-06-07-08-09-0A-0B-0C-0D-0E-0F       Text%N")
			line_out ("Dec-Addr=00-01-02-03-04-05-06-07-08-09-10-11-12-13-14-15 0123456789ABCDEF%N")
			line_out ("======== =============================================== ================%N")
			line_out ("00000000 ")
			across
				1 |..| l_file.count as ic
			loop
				l_file.read_character
				l_hex := l_file.last_character.code.to_hex_string.as_string_32
				l_hex.remove_head (6)
				line_out (l_hex.as_string_8)
				if (ic.item \\ 16) = 0 then
					line_out (" ")
					l_text.append_character (l_file.last_character.to_character_32)
					l_text.replace_substring_all ("%N".as_string_32, "".as_string_32)
					line_out (l_text.as_string_8)
					create l_text.make_empty
					line_out ("(@" + i.out + ")%N")
					l_hex := ic.item.to_hex_string.as_string_32
					line_out (l_hex + " ".as_string_8)
				else
					line_out ("-")
					l_text.append_character (l_file.last_character.to_character_32)
				end
				l_hex := l_file.last_character.code.to_hex_string.as_string_32
				l_hex.remove_head (6)
				l_hex.append_character ('-'.to_character_32)
				i := ic.item - 1
			end
			line_out ("EOF%N")
			l_file.close
			check
					attached out_file as al_out_file
			then
				al_out_file.close
			end
		end

	out_file: detachable PLAIN_TEXT_FILE
			-- DBF output file.
		attribute
			create Result.make_create_read_write (Out_file_name)
		end

	Out_file_name: STRING_8 = "dbf_content.txt"

	line_out (a_line: STRING_8)
		do
			check
					attached out_file as al_out_file
			then
				al_out_file.put_string (a_line)
			end
			print (a_line)
		end
